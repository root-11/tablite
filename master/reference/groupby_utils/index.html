    def update(self, value):
    if isinstance(value, (date, time, datetime, str)):
        raise ValueError(f"Std.dev. of {type(value)} doesn't make sense.")
    if value is not None:
        self.count += 1
        dt = value - self.mean
        self.mean += dt / self.count
        self.c += dt * (value - self.mean)
